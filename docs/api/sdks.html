npm

npm install @pitchey/sdk

yarn

yarn add @pitchey/sdk

pnpm

pnpm add @pitchey/sdk

Quick Start

import { PitcheySDK } from '@pitchey/sdk'; const pitchey = new PitcheySDK(); await pitchey.auth.login('user@example.com', 'password'); const pitches = await pitchey.pitches.getPublic({ limit: 10 });

Python Installation

pip

pip install pitchey-sdk

With data analysis tools

pip install pitchey-sdk[analysis]

Quick Start

import asyncio from pitchey_sdk import PitcheySDK async def main(): async with PitcheySDK() as pitchey: await pitchey.auth.login('user@example.com', 'password') pitches = await pitchey.pitches.get_public(limit=10) print(f"Found {len(pitches['data'])} pitches") asyncio.run(main())

Usage Examples

TypeScript/JavaScript

// React component example import { useState, useEffect } from 'react'; import { PitcheySDK } from '@pitchey/sdk'; function PitchList() { const [pitches, setPitches] = useState([]); useEffect(() => { const loadPitches = async () => { const pitchey = PitcheySDK.public(); const result = await pitchey.pitches.getPublic(); setPitches(result.data); }; loadPitches(); }, []); return ( <div> {pitches.map(pitch => ( <div key={pitch.id}>{pitch.title}</div> ))} </div> ); }

Python Data Analysis

# Data analysis example import pandas as pd from pitchey_sdk import PitcheySDK async def analyze_market(): async with PitcheySDK.public() as pitchey: pitches = await pitchey.pitches.get_public(limit=1000) df = pd.DataFrame(pitches['data']) # Analyze by genre genre_stats = df.groupby('genre').agg({ 'view_count': ['mean', 'sum'], 'like_count': 'mean' }) print(genre_stats) return df
